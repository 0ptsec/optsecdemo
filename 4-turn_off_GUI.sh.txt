#!/bin/bash

echo "$(tput setaf 3)$(tput bold)===============================Running tirn_off_GUI.sh=================================$(tput sgr0)"

sleep 5

sudo echo 'gpu_mem=16' | sudo tee -a /boot/config.txt >/dev/null
sudo update-rc.d lightdm disable
echo "$(tput setaf 1)$(tput bold)Disable desktop in System Options ---> Boot / Auto Logon by selecting B1 - Text console, requiring user to logon.$(tput sg$
echo "$(tput setaf 1)$(tput bold)You'll be asked to reboot afterwards, say yes.$(tput sgr0)"
read -n 1 -s -r -p "After reading, press any key to continue"
sudo raspi-config

echo "$(tput setaf 3)$(tput bold)=====================done======================$(tput sgr0)"
